Os comandos abaixo, são para criar a pasta .env, activar a pasta .env e correr o projecto
##########################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################################
#Criar a pasta .env
    => python -m venv .venv

#Para poder aceder ao servidor Windows
    => Set-ExecutionPolicy RemoteSigned -Scope CurrentUser

#Activar a pasta .env
    #Para Windows
    => .venv/Scripts/Activate

    #Para Linux
    => . .venv/bin/activate

#Para instalar as dependencias
    => pip install -r requirements.txt

#Para correr a aplicação
    => flask run --port 5500
