export const fetchChatbotResponse = async (query, accessCode) => {
  const API_URL = 'http://10.53.21.102:5000/api/chatbot';
  
  try {
    const response = await fetch(API_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      // Backend agora aceita múltiplas perguntas com o mesmo código
      body: JSON.stringify({ 
        query, 
        access_code: accessCode 
      }),
    });

    if (response.status === 403) {
      const errorData = await response.json();
      throw new Error(errorData.message || 'Código PREMIUM inválido ou expirado.');
    }

    if (response.status === 401) {
      throw new Error('O código de acesso PREMIUM é obrigatório.');
    }

    if (!response.ok) {
      const errorData = await response.json();
      throw new Error(errorData.message || 'Erro na resposta do servidor');
    }

    return await response.json();
  } catch (error) {
    console.error('Erro ao chamar API do chatbot:', error);
    throw error;
  }
};
